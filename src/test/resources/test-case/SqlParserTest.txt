[sql-037:where]
SELECT t.a,d,max(b) as max_b,sum(c) as sum_c
FROM t
WHERE t.a IS NOT NULL
  and b like '%'
  and (c > 1 or c < 1)
  and not ( c >= 1 and c <= 1)
  and c != 1
  and c =1
  and c in (1,2,3)
  and c between 1 and 3
group by t.a,d
having max(b)>100 and t.a like 'a%'
order by max(t.b) desc, sum(c) asc

  x,sum(y) as y
from t
left join t1
  on 1=1
right join t2
  on 1>2 and 1<3

group     by 1,       f(x)
order by 1 asc, 2 desc
